// Calculate item amount: quantity Ã— rate
export const calculateItemAmount = (quantity: number, rate: number): number => {
  return quantity * rate;
};

// Calculate document totals including VAT
export const calculateDocumentTotals = (
  items: DocumentItem[], 
  vatRate: number = 0.18
): { subtotal: number; vat: number; total: number } => {
  const subtotal = items.reduce((sum, item) => sum + item.amount, 0);
  const vat = subtotal * vatRate;
  const total = subtotal + vat;
  
  return { subtotal, vat, total };
};

4. File Download
typescript
const downloadDocument = (document: Document) => {
  // Create JSON blob
  const blob = new Blob([JSON.stringify(document, null, 2)], {
    type: 'application/json'
  });
  
  // Create download link
  const url = URL.createObjectURL(blob);
  const a = document.createElement('a');
  a.href = url;
  a.download = `${document.type}-${document.documentNo}.json`;
  
  // Trigger download
  document.body.appendChild(a);
  a.click();
  document.body.removeChild(a);
  URL.revokeObjectURL(url);
};
